\section{Problem Statement}
%
\begin{problem}
Generate
\begin{equation}
x(n)=\sum_{k=1}^{3}h(k)b_{n-k}+v(n)
\end{equation}
%
where 
%
\begin{equation}
h(n)=\begin{cases}
\frac{1}{2}\sbrak{1+\cos\brak{\frac{2\pi}{F}{(n-2)}}}& n=1,2,3\\
0 & \text{otherwise}
\end{cases},
\end{equation}
%
$b_n=\pm 1$ and $v(n) \sim \mathcal{N}\brak{0,\sigma^2}$ for SNR ranging from 0 to 10 dB.
\end{problem}
%
\begin{problem}
Formulate the cost function for the LMS algorithm.
\end{problem}
%
\solution
\begin{align}
e(n) = b(n-M)-W(n)^TX(n)
\end{align}
%
where all vectors on the R.H.S are of length $M$ and the entries of $X(n)$ are obtained from
$x(n)$. The problem can then be expressed as
\begin{equation}
\min_We^2(n)
\end{equation}
